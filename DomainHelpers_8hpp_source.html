145 /// \ingroup ComputationalDomainGroup
146 /// Iterates over the corners of a VolumeDim-dimensional cube.
147 template <size_t VolumeDim>
148 class VolumeCornerIterator {
149  public:
150  VolumeCornerIterator() noexcept = default;
151  explicit VolumeCornerIterator(size_t initial_local_corner_number) noexcept
152  : local_corner_number_(initial_local_corner_number) {}
153  VolumeCornerIterator(
154  // The block index is also global corner
155  // index of the lowest corner of the block.
156  Index<VolumeDim> block_index,
157  Index<VolumeDim> global_corner_extents) noexcept
158  : global_corner_number_(
159  collapsed_index(block_index, global_corner_extents)),
160  global_corner_index_(block_index),
161  global_corner_extents_(global_corner_extents) {}
162  void operator++() noexcept {
163  ++local_corner_number_;
164  for (size_t i = 0; i < VolumeDim; i++) {
165  gsl::at(coords_of_corner_, i) =
166  2.0 * get_nth_bit(local_corner_number_, i) - 1.0;
167  gsl::at(array_sides_, i) =
168  2 * get_nth_bit(local_corner_number_, i) - 1 == 1 ? Side::Upper
169  : Side::Lower;
170  }
171  }
172  explicit operator bool() const noexcept {
173  return local_corner_number_ < two_to_the(VolumeDim);
174  }
175  const size_t& local_corner_number() const noexcept {
176  return local_corner_number_;
177  }
178  size_t global_corner_number() const noexcept {
179  std::array<size_t, VolumeDim> new_indices{};
180  for (size_t i = 0; i < VolumeDim; i++) {
181  gsl::at(new_indices, i) =
182  global_corner_index_[i] +
183  (gsl::at(array_sides_, i) == Side::Upper ? 1 : 0);
184  }
185  const Index<VolumeDim> interior_multi_index(new_indices);
186  return collapsed_index(interior_multi_index, global_corner_extents_);
187  }
188  const std::array<Side, VolumeDim>& operator()() const noexcept {
189  return array_sides_;
190  }
191  const std::array<Side, VolumeDim>& operator*() const noexcept {
192  return array_sides_;
193  }
194  const std::array<double, VolumeDim>& coords_of_corner() const noexcept {
195  return coords_of_corner_;
196  }
197 
198  private:
199  size_t local_corner_number_ = 0;
200  size_t global_corner_number_{std::numeric_limits<size_t>::max()};
201  Index<VolumeDim> global_corner_index_{};
202  Index<VolumeDim> global_corner_extents_{};
203  std::array<Side, VolumeDim> array_sides_ = make_array<VolumeDim>(Side::Lower);
204  std::array<double, VolumeDim> coords_of_corner_ = make_array<VolumeDim>(-1.0);
205 };
