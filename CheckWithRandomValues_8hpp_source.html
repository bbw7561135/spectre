98 template <class F, class T, class TagsList, class Klass, class... ReturnTypes,
99  class... ArgumentTypes, class... MemberArgs, size_t... ResultIs,
100  size_t... ArgumentIs, size_t... MemberArgsIs>
101 void check_with_random_values_impl(
102  F&& f, const Klass& klass, const std::string& module_name,
103  const std::vector<std::string>& function_names, std::mt19937 generator,
104  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
105  distributions,
106  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
107  tmpl::list<ReturnTypes...> /*return_types*/,
108  tmpl::list<ArgumentTypes...> /*argument_types*/,
109  std::index_sequence<ResultIs...> /*index_return_types*/,
110  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
111  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
112  TagsList /*meta*/) {
113  // Note: generator and distributions cannot be const.
114  std::tuple<ArgumentTypes...> args{
115  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
116  // We fill with random values after initialization because the order of
117  // evaluation is not guaranteed for a constructor call and so then knowing
118  // the seed would not lead to reproducible results.
119  (void)std::initializer_list<char>{(
120  (void)fill_with_random_values(
121  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
122  make_not_null(&(distributions[ArgumentIs]))),
123  '0')...};
124 
125  size_t count = 0;
126  tmpl::for_each<TagsList>([&f, &klass, &args, &used_for_size, &member_args,
127  &module_name, &function_names, &count](auto tag) {
128  (void)used_for_size; // Avoid compiler warning
129  using Tag = tmpl::type_from<decltype(tag)>;
130  const auto result =
131  tuples::get<Tag>((klass.*f)(std::get<ArgumentIs>(args)...));
132  CHECK_ITERABLE_APPROX(
133  result,
134  (pypp::call<std::decay_t<decltype(result)>>(
135  module_name, function_names[count], std::get<ArgumentIs>(args)...,
136  forward_to_pypp<std::decay_t<decltype(result)>>(
137  std::get<MemberArgsIs>(member_args), used_for_size)...)));
138  count++;
139  });
140 }

142 template <class F, class T, class Klass, class... ArgumentTypes,
143  class... MemberArgs, size_t... ArgumentIs, size_t... MemberArgsIs>
144 void check_with_random_values_impl(
145  F&& f, const Klass& klass, const std::string& module_name,
146  const std::string& function_name, std::mt19937 generator,
147  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
148  distributions,
149  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
150  tmpl::list<ArgumentTypes...> /*argument_types*/,
151  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
152  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
153  NoSuchType /*meta*/) {
154  // Note: generator and distributions cannot be const.
155  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
156  using ResultType = typename f_info::return_type;
157  std::tuple<ArgumentTypes...> args{
158  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
159  // We fill with random values after initialization because the order of
160  // evaluation is not guaranteed for a constructor call and so then knowing the
161  // seed would not lead to reproducible results.
162  (void)std::initializer_list<char>{(
163  (void)fill_with_random_values(
164  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
165  make_not_null(&(distributions[ArgumentIs]))),
166  '0')...};
167  const auto result = make_overloader(
168  [&](std::true_type /*is_class*/, auto&& local_f) {
169  return (klass.*local_f)(std::get<ArgumentIs>(args)...);
170  },
171  [&](std::false_type /*is_class*/, auto&& local_f) {
172  return local_f(std::get<ArgumentIs>(args)...);
173  })(
174  std::integral_constant<
175  bool, not cpp17::is_same_v<NoSuchType, std::decay_t<Klass>>>{},
176  std::forward<F>(f));
177  CHECK_ITERABLE_APPROX(
178  result, pypp::call<ResultType>(
179  module_name, function_name, std::get<ArgumentIs>(args)...,
180  forward_to_pypp<ResultType>(
181  std::get<MemberArgsIs>(member_args), used_for_size)...));
182 }

184 template <class F, class T, class Klass, class... ReturnTypes,
185  class... ArgumentTypes, class... MemberArgs, size_t... ResultIs,
186  size_t... ArgumentIs, size_t... MemberArgsIs>
187 void check_with_random_values_impl(
188  F&& f, const Klass& klass, const std::string& module_name,
189  const std::vector<std::string>& function_names, std::mt19937 generator,
190  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
191  distributions,
192  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
193  tmpl::list<ReturnTypes...> /*return_types*/,
194  tmpl::list<ArgumentTypes...> /*argument_types*/,
195  std::index_sequence<ResultIs...> /*index_return_types*/,
196  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
197  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
198  NoSuchType /* meta */) {
199  // Note: generator and distributions cannot be const.
200  std::tuple<ReturnTypes...> results{
201  make_with_value<ReturnTypes>(used_for_size, 0.0)...};
202  std::tuple<ArgumentTypes...> args{
203  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
204  // We fill with random values after initialization because the order of
205  // evaluation is not guaranteed for a constructor call and so then knowing the
206  // seed would not lead to reproducible results.
207  (void)std::initializer_list<char>{(
208  (void)fill_with_random_values(
209  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
210  make_not_null(&(distributions[ArgumentIs]))),
211  '0')...};
212  // We intentionally do not set the return value to signaling NaN so that not
213  // all of our functions need to be able to handle the cases where they
214  // receive a NaN. Instead, we fill the return value with random numbers.
215  (void)std::initializer_list<char>{(
216  (void)fill_with_random_values(make_not_null(&std::get<ResultIs>(results)),
217  make_not_null(&generator),
218  make_not_null(&(distributions[0]))),
219  '0')...};
220  make_overloader(
221  [&](std::true_type /*is_class*/, auto&& local_f) {
222  (klass.*local_f)(make_not_null(&std::get<ResultIs>(results))...,
223  std::get<ArgumentIs>(args)...);
224  },
225  [&](std::false_type /*is_class*/, auto&& local_f) {
226  local_f(make_not_null(&std::get<ResultIs>(results))...,
227  std::get<ArgumentIs>(args)...);
228  })(
229  std::integral_constant<
230  bool, not cpp17::is_same_v<NoSuchType, std::decay_t<Klass>>>{},
231  std::forward<F>(f));
232  const auto helper = [&module_name, &function_names, &args, &results,
233  &member_args, &used_for_size](auto result_i) {
234  (void)member_args; // avoid compiler warning
235  (void)used_for_size; // avoid compiler warning
236  constexpr size_t iter = decltype(result_i)::value;
237  CHECK_ITERABLE_APPROX(
238  std::get<iter>(results),
239  (pypp::call<std::tuple_element_t<iter, std::tuple<ReturnTypes...>>>(
240  module_name, function_names[iter], std::get<ArgumentIs>(args)...,
241  forward_to_pypp<
242  std::tuple_element_t<iter, std::tuple<ReturnTypes...>>>(
243  std::get<MemberArgsIs>(member_args), used_for_size)...)));
244  return '0';
245  };
246  (void)std::initializer_list<char>{
247  helper(std::integral_constant<size_t, ResultIs>{})...};
248 }

251 /*!
252  * \brief Tests a C++ function returning by value by comparing the result to a
253  * python function
254  *
255  * Tests the function `f` by comparing the result to that of the python function
256  * `function_name` in the file `module_name`. The function is tested by
257  * generated random values in the half-open range [`lower_bound`,
258  * `upper_bound`). The argument `used_for_size` is used for constructing the
259  * arguments of `f` by calling `make_with_value<ArgumentType>(used_for_size,
260  * 0.0)`.
261  *
262  * \note You must explicitly pass the number of bounds you will be passing as
263  * the first template parameter, the rest will be inferred.
264  *
265  * \note If you have a test fail you can replay the scenario by feeding in the
266  * seed that was printed out in the failed test as the last argument.
267  *
268  * \param f The C++ function to test
269  * \param module_name The python file relative to the directory used in
270  * `SetupLocalPythonEnvironment`
271  * \param function_name The name of the python function inside `module_name`
272  * \param lower_and_upper_bounds The lower and upper bounds for the randomly
273  * generated numbers. Must be either an array of a single pair, or of as many
274  * pairs as there are arguments to `f` that are not a `gsl::not_null`
275  * \param used_for_size The type `X` for the arguments of `f` of type
276  *`Tensor<X>`
277  * \param seed The seed for the random number generator. This should only be
278  * specified when debugging a failure with a particular set of random numbers,
279  * in general it should be left to the default value.
280  */
281 template <size_t NumberOfBounds, class F, class T,
282  Requires<not cpp17::is_same_v<
283  typename tt::function_info<cpp20::remove_cvref_t<F>>::return_type,
284  void>> = nullptr>
285 // The Requires is used so that we can call the std::vector<std::string> with
286 // braces and not have it be ambiguous.
287 void check_with_random_values(
288  F&& f, const std::string& module_name, const std::string& function_name,
289  const std::array<std::pair<double, double>, NumberOfBounds>&
290  lower_and_upper_bounds,
291  const T& used_for_size,
292  const typename std::random_device::result_type seed =
293  std::random_device{}()) {
294  INFO("seed: " << seed);
295  std::mt19937 generator(seed);
296  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
297  using number_of_not_null =
298  tmpl::count_if<typename f_info::argument_types,
299  tmpl::bind<TestHelpers_detail::is_not_null, tmpl::_1>>;
300  using argument_types = tmpl::transform<
301  tmpl::pop_front<typename f_info::argument_types, number_of_not_null>,
302  std::decay<tmpl::_1>>;
303 
304  static_assert(number_of_not_null::value == 0,
305  "Cannot return arguments by gsl::not_null if the python "
306  "function name is passed as a string. If the function only "
307  "returns one gsl::not_null then you must pass in a one element "
308  "vector<string>.");
309  static_assert(tmpl::size<argument_types>::value != 0,
310  "The function 'f' must take at least one argument.");
311  static_assert(NumberOfBounds == 1 or
312  NumberOfBounds == tmpl::size<argument_types>::value,
313  "The number of lower and upper bound pairs must be either 1 or "
314  "equal to the number of arguments taken by f that are not "
315  "gsl::not_null.");
316  std::array<std::uniform_real_distribution<>,
317  tmpl::size<argument_types>::value>
318  distributions;
319  for (size_t i = 0; i < tmpl::size<argument_types>::value; ++i) {
320  gsl::at(distributions,
321  NumberOfBounds == 1 ? 0 : i) = std::uniform_real_distribution<>{
322  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).first,
323  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).second};
324  }
325  TestHelpers_detail::check_with_random_values_impl(
326  std::forward<F>(f), NoSuchType{}, module_name, function_name, generator,
327  std::move(distributions), std::tuple<>{}, used_for_size, argument_types{},
328  std::make_index_sequence<tmpl::size<argument_types>::value>{},
329  std::make_index_sequence<0>{}, NoSuchType{});
330 }

332 /*!
333  * \brief Tests a C++ function returning by `gsl::not_null` by comparing the
334  * result to a python function
335  *
336  * Tests the function `f` by comparing the result to that of the python function
337  * `function_name` in the file `module_name`. The function is tested by
338  * generated random values in the half-open range [`lower_bound`, `upper_bound`)
339  * for each argument. The argument `used_for_size` is used for constructing the
340  * arguments of `f` by calling `make_with_value<ArgumentType>(used_for_size,
341  * 0.0)`. For functions that return by `gsl::not_null`, the result will be
342  * initialized with random values rather than to signaling `NaN`s. This means
343  * functions do not need to support receiving a signaling `NaN` in their return
344  * argument to be tested using this function.
345  *
346  * \note You must explicitly pass the number of bounds you will be passing as
347  * the first template parameter, the rest will be inferred.
348  *
349  * \note If you have a test fail you can replay the scenario by feeding in the
350  * seed that was printed out in the failed test as the last argument.
351  *
352  * \param f The C++ function to test
353  * \param module_name The python file relative to the directory used in
354  * `SetupLocalPythonEnvironment`
355  * \param function_names The names of the python functions inside `module_name`
356  * in the order that they return the `gsl::not_null` results
357  * \param lower_and_upper_bounds The lower and upper bounds for the randomly
358  * generated numbers. Must be either an array of a single pair, or of as many
359  * pairs as there are arguments to `f` that are not a `gsl::not_null`
360  * \param used_for_size The type `X` for the arguments of `f` of type
361  * `Tensor<X>`
362  * \param seed The seed for the random number generator. This should only be
363  * specified when debugging a failure with a particular set of random numbers,
364  * in general it should be left to the default value.
365  */
366 template <size_t NumberOfBounds, class F, class T>
367 void check_with_random_values(
368  F&& f, const std::string& module_name,
369  const std::vector<std::string>& function_names,
370  const std::array<std::pair<double, double>, NumberOfBounds>&
371  lower_and_upper_bounds,
372  const T& used_for_size,
373  const typename std::random_device::result_type seed =
374  std::random_device{}()) {
375  INFO("seed: " << seed);
376  std::mt19937 generator(seed);
377  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
378  using number_of_not_null =
379  tmpl::count_if<typename f_info::argument_types,
380  tmpl::bind<TestHelpers_detail::is_not_null, tmpl::_1>>;
381  using argument_types = tmpl::transform<
382  tmpl::pop_front<typename f_info::argument_types, number_of_not_null>,
383  std::decay<tmpl::_1>>;
384  using return_types =
385  tmpl::transform<tmpl::pop_back<typename f_info::argument_types,
386  tmpl::size<argument_types>>,
387  TestHelpers_detail::RemoveNotNull<tmpl::_1>>;
388 
389  static_assert(number_of_not_null::value != 0,
390  "You must return at least one argument by gsl::not_null when "
391  "passing the python function names as a vector<string>. If "
392  "your function returns by value do not pass the function name "
393  "as a vector<string> but just a string.");
394  static_assert(
395  cpp17::is_same_v<typename f_info::return_type, void>,
396  "A function returning by gsl::not_null must have a void return type.");
397  static_assert(tmpl::size<argument_types>::value != 0,
398  "The function 'f' must take at least one argument.");
399  static_assert(NumberOfBounds == 1 or
400  NumberOfBounds == tmpl::size<argument_types>::value,
401  "The number of lower and upper bound pairs must be either 1 or "
402  "equal to the number of arguments taken by f that are not "
403  "gsl::not_null.");
404  if (function_names.size() != number_of_not_null::value) {
405  ERROR(
406  "The number of python functions passed must be the same as the number "
407  "of gsl::not_null arguments in the C++ function. The order of the "
408  "python functions must also be the same as the order of the "
409  "gsl::not_null arguments.");
410  }
411  std::array<std::uniform_real_distribution<>,
412  tmpl::size<argument_types>::value>
413  distributions;
414  for (size_t i = 0; i < tmpl::size<argument_types>::value; ++i) {
415  gsl::at(distributions,
416  NumberOfBounds == 1 ? 0 : i) = std::uniform_real_distribution<>{
417  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).first,
418  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).second};
419  }
420  TestHelpers_detail::check_with_random_values_impl(
421  std::forward<F>(f), NoSuchType{}, module_name, function_names, generator,
422  std::move(distributions), std::tuple<>{}, used_for_size, return_types{},
423  argument_types{},
424  std::make_index_sequence<tmpl::size<return_types>::value>{},
425  std::make_index_sequence<tmpl::size<argument_types>::value>{},
426  std::make_index_sequence<0>{}, NoSuchType{});
427 }

458  * should have the same types as the normal arguments passed to the member

460  * \param used_for_size The type `X` for the arguments of `f` of type

479  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,

485  using number_of_not_null =

520 /*!
521  * \brief Tests a member function of a class returning by either `gsl::not_null`
522  * or `TaggedTuple` by comparing the result to a python function
523  *
524  * Tests the function `f` by comparing the result to that of the python
525  * functions `function_names` in the file `module_name`. An instance of the
526  * class is passed in as the second argument and is the object on which the
527  * member function `f` will be invoked. The member function is invoked as
528  * `klass.function`, so passing in pointers is not supported. The function is
529  * tested by generated random values in the half-open range [`lower_bound`,
530  * `upper_bound`). The argument `used_for_size` is used for constructing the
531  * arguments of `f` by calling `make_with_value<ArgumentType>(used_for_size,
532  * 0.0)`. For functions that return by `gsl::not_null`, the result will be
533  * initialized with random values rather than to signaling `NaN`s. This means
534  * functions do not need to support receiving a signaling `NaN` in their return
535  * argument to be tested using this function.
536  *
537  * If `TagsList` is passed as a `tmpl::list`, then `f` is expected to
538  * return a TaggedTuple. The result of each python function will be
539  * compared with calling `tuples::get` on the result of `f`. The order of the
540  * tags within `TagsList` should match the order of the functions in
541  * `function_names`
542  *
543  * \note You must explicitly pass the number of bounds you will be passing as
544  * the first template parameter, the rest will be inferred.
545  *
546  * \note If you have a test fail you can replay the scenario by feeding in the
547  * seed that was printed out in the failed test as the last argument.
548  *
549  * \param f The member function to test
550  * \param klass the object on which to invoke `f`
551  * \param module_name The python file relative to the directory used in
552  * `SetupLocalPythonEnvironment`
553  * \param function_names The names of the python functions inside `module_name`
554  * in the order that they return the `gsl::not_null` results
555  * \param lower_and_upper_bounds The lower and upper bounds for the randomly
556  * generated numbers. Must be either an array of a single pair, or of as many
557  * pairs as there are arguments to `f` that are not a `gsl::not_null`
558  * \param member_args a tuple of the member variables of the object `klass` that
559  * the python function will need in order to perform the computation. These
560  * should have the same types as the normal arguments passed to the member
561  * function, e.g. `Tensor<X>`.
562  * \param used_for_size The type `X` for the arguments of `f` of type
563  * `Tensor<X>`
564  * \param seed The seed for the random number generator. This should only be
565  * specified when debugging a failure with a particular set of random numbers,
566  * in general it should be left to the default value.
567  */
568 template <size_t NumberOfBounds, typename TagsList = NoSuchType, class F,
569  class T, class... MemberArgs>
570 void check_with_random_values(
571  F&& f,
572  const typename tt::function_info<cpp20::remove_cvref_t<F>>::class_type&
573  klass,
574  const std::string& module_name,
575  const std::vector<std::string>& function_names,
576  const std::array<std::pair<double, double>, NumberOfBounds>&
577  lower_and_upper_bounds,
578  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
579  const typename std::random_device::result_type seed =
580  std::random_device{}()) {
581  INFO("seed: " << seed);
582  std::mt19937 generator(seed);
583  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
584  using number_of_not_null =
585  tmpl::count_if<typename f_info::argument_types,
586  tmpl::bind<TestHelpers_detail::is_not_null, tmpl::_1>>;
587  using argument_types = tmpl::transform<
588  tmpl::pop_front<typename f_info::argument_types, number_of_not_null>,
589  std::decay<tmpl::_1>>;
590  using return_types =
591  tmpl::transform<tmpl::pop_back<typename f_info::argument_types,
592  tmpl::size<argument_types>>,
593  TestHelpers_detail::RemoveNotNull<tmpl::_1>>;
594 
595  static_assert(
596  number_of_not_null::value != 0 or tt::is_a_v<tmpl::list, TagsList>,
597  "You must either return at least one argument by gsl::not_null when"
598  "passing the python function names as a vector<string>, or return by "
599  "value using a TaggedTuple. If your function returns by value with a "
600  "type that is not a TaggedTuple do not pass the function name as a "
601  "vector<string> but just a string.");
602  static_assert(cpp17::is_same_v<typename f_info::return_type, void> or
603  tt::is_a_v<tmpl::list, TagsList>,
604  "The function must either return by gsl::not_null and have a "
605  "void return type, or return by TaggedTuple");
606  static_assert(tmpl::size<argument_types>::value != 0,
607  "The function 'f' must take at least one argument.");
608  static_assert(NumberOfBounds == 1 or
609  NumberOfBounds == tmpl::size<argument_types>::value,
610  "The number of lower and upper bound pairs must be either 1 or "
611  "equal to the number of arguments taken by f that are not "
612  "gsl::not_null.");
613  if (function_names.size() != number_of_not_null::value and
614  not tt::is_a_v<tmpl::list, TagsList>) {
615  ERROR(
616  "If testing a function that returns by gsl::not_null, the number of "
617  "python functions passed must be the same as the number of "
618  "gsl::not_null arguments in the C++ function. The order of the "
619  "python functions must also be the same as the order of the "
620  "gsl::not_null arguments.");
621  }
622  std::array<std::uniform_real_distribution<>,
623  tmpl::size<argument_types>::value>
624  distributions;
625  for (size_t i = 0; i < tmpl::size<argument_types>::value; ++i) {
626  gsl::at(distributions,
627  NumberOfBounds == 1 ? 0 : i) = std::uniform_real_distribution<>{
628  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).first,
629  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).second};
630  }
631  TestHelpers_detail::check_with_random_values_impl(
632  std::forward<F>(f), klass, module_name, function_names, generator,
633  std::move(distributions), member_args, used_for_size, return_types{},
634  argument_types{},
635  std::make_index_sequence<tmpl::size<return_types>::value>{},
636  std::make_index_sequence<tmpl::size<argument_types>::value>{},
637  std::make_index_sequence<sizeof...(MemberArgs)>{}, TagsList{});
638 }
